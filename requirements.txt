# Oxillia Frontend - System & Project Requirements
# This file documents all system requirements, dependencies, and configurations needed to run the project.

# ============================================================================
# SYSTEM REQUIREMENTS
# ============================================================================

# Operating System
# - Windows 10+, macOS 10.15+, or Linux (Ubuntu 20.04+)

# Runtime Environment
# - Node.js: 18.x.x or higher (LTS recommended)
# - npm: 9.x.x or higher
# - Git: 2.30+

# Command to verify installations:
# - node --version    (should show v18.x.x or higher)
# - npm --version     (should show 9.x.x or higher)
# - git --version     (should show 2.30+)

# ============================================================================
# DEVELOPMENT ENVIRONMENT SETUP
# ============================================================================

# 1. Install Node.js with npm
#    Download from: https://nodejs.org/
#    Recommended: LTS version (18.x.x or 20.x.x)

# 2. Install Angular CLI globally (optional but recommended)
npm install -g @angular/cli@20.3.13

# 3. Clone the repository
git clone <repository-url>
cd Oxillia-frontend

# 4. Install project dependencies
npm install

# ============================================================================
# TROUBLESHOOTING
# ============================================================================

# Issue: Port 4200 already in use
# Solution: ng serve --port 4201

# Issue: npm ERR! code ERESOLVE
# Solution: npm install --legacy-peer-deps

# Issue: Out of memory during build
# Solution: node --max_old_space_size=4096 node_modules/@angular/cli/bin/ng build

# Issue: Module not found errors
# Solution: 
#   rm -rf node_modules
#   rm package-lock.json
#   npm install
